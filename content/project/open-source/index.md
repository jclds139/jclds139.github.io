    function setRepoCardsTheme(is_dark) {
        const local_cards = document.getElementsByClassName("repo-card");

            if (is_dark)
                local_cards[i].setAttribute('data-theme', 'dark-theme');

                    local_cards[i].removeAttribute('data-theme');
        }
    };

    });
